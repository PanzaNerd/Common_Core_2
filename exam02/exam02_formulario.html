🔡 maiuscole / minuscole
minuscolo → maiuscolo

maiuscolo → minuscolo
c = c + 32;   // oppure c - 'A' + 'a'
rot13 — prima metà +13, seconda metà -13
if (c >= 'a' && c <= 'm') c += 13;
if (c >= 'n' && c <= 'z') c -= 13;
if (c >= 'A' && c <= 'M') c += 13;
if (c >= 'N' && c <= 'Z') c -= 13;
alpha_mirror — specchio alfabeto
c = 'a' + ('z' - c);   // minuscolo
c = 'A' + ('Z' - c);   // maiuscolo
str_capitalizer — prima lettera parola maiuscola
int new_word = 1;
while (str[i])
{
    if (str[i] == ' ' || str[i] == '\t')
        new_word = 1;
    else if (new_word == 1)
    {
        if (str[i] >= 'a' && str[i] <= 'z')
            str[i] = str[i] - 32;
        new_word = 0;
    }
    else if (str[i] >= 'A' && str[i] <= 'Z')
        str[i] = str[i] + 32;
    write(1, &str[i], 1);
    i++;
}
